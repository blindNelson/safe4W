create table UsuarioDenuncia (
    idUsuarioDenuncia,
    idUsuario,
    idDenuncia,
    PRIMARY KEY (idUsuarioDenuncia),
    FOREIGN KEY (idUsuario) references Usuario(idUsuario),
    FOREIGN KEY (idDenuncia) references Denuncia(idDenuncia)
)

Create table Denuncia(
	idDenuncia int not null,
	idRegião int not null,
	nomeVitima varChar(30) not null,
	nomeAgressor varChar(30) not null,
	dataDenuncia DATE not null,
	PRYMARY KEY (idDenuncia),
	FOREIGN KEY (idRegião) REFERENCES Região(idRegião)
)

create table Região(
	idRegião int not null,
	nomeRegião varchar(20) null,
	PRIMARY KEY (idRegião)
)
